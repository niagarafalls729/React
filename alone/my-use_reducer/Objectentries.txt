Object.entries()메서드

Object.entries()메서드는 순서가 매겨진 객체의 [key , value ]쌍을 배열로 반환.


const obj= {

    1: '홍길동',
    2: 20,
    3: '한국'
}
1 정렬없이
 

for(const[k,v] of Object.entries(obj)){
    console.log("k  "+`${k} V: ${v}`)

    console.log(k+"  "+ v)
}


 
2 정렬 
a 안에 0은 키값이고 뒤에가 값이니깐
for(const[k,v] of Object.entries(obj).sort((a,b) => a[0].localeCompare(b[0]))){
    console.log("k  "+`${k} V: ${v}`)

    console.log(k+"  "+ v)
}
